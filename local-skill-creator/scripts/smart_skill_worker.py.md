# Example 1: Create new skill from scratch

  "Create a skill to fetch Amazon bestseller data for any category" \
  --output-dir ~/.openclaw/skills

# Example 2: Fix existing skill with low pass rate
python3 smart_skill_worker.py \
  "Fix amazon-scraper skill to handle rate limiting and timeout errors" \
  --output-dir ~/.openclaw/skills/amazon-scraper \
  --max-iterations 3

# Example 3: Optimize for higher quality
python3 smart_skill_worker.py \
  "Optimize feishu-doc-edit to reduce token usage and improve image positioning accuracy" \
  --target-pass-rate 0.95 \
  --max-iterations 10

# Example 4: Use with existing skill (update workflow)
# This script will:
# 1. Read existing SKILL.md
# 2. Run evals if evals/evals.json exists
# 3. If failed/unsatisfied → LLM fixes + optimizes
# 4. Loop until pass_rate >= target (default 90%)
# 5. Automatically publish to GitHub + MemOS queue


# Step-by-step breakdown (what happens internally):

# 1. CREATE
# └── init_skill.py generates SKILL.md with anthropics tests section
# └── Creates evals/ directory with default evals

# 2. FIRST TEST
# └── run_tests.py spawns with-skill + baseline (parallel)
# └── Captures timing.json (tokens, duration_ms)
# └── Grading.json generated by assertions.py

# 3. ANALYZE
# └── Calculate pass_rate (passed / total assertions)
# └── Check if pass_rate >= target (default 90%)

# 4. FIX (if pass_rate < target)
# └── LLM reads error logs + grading.json
# └── Generates bug fix for SKILL.md
# └── Overwrites SKILL.md with fixed version

# 5. OPTIMIZE (if still pass_rate < target)
# └── LLM analyzes low pass rate
# └── Generates optimization suggestions
# └── Improves clarity, examples, error handling

# 6. RETRY
# └── Loop back to Step 2 until:
#     ✓ pass_rate >= target
#     OR
#     ✓ max_iterations reached

# 7. PUBLISH
# └── publish_skill.py runs:
#     ├── quick_validate.py (format check)
#     ├── git add + commit + push
#     └── write to ~/.openclaw/skills/.pending-memos.json

# 8. REGISTER (AI agent auto-registers)
# └── memory_write_public(content, summary)
# └──.skill available for skill_search()


# What you get after success:

# File structure:
# ~/.openclaw/skills/my-automated-skill/
# ├── SKILL.md (optimized, auto-generated tests section)
# ├── evals/
# │   ├── evals.json (test cases)
# │   └── assertions.py (grading logic)
# ├── README.md (auto-generated)
# └── scripts/
